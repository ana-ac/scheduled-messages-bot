// requirements
const config = require('./config/config');
const helpers = require('./../bin/utils');

/**
 * Bot Class is the father of the implemented bots and handle the common logic
 * @class
 * @see {@link bot.js}
 */
class Bot {
    /**
     *
     * @param {object} bot
     */
    constructor(bot) {
        this.bot = bot;
        this.loadConfig();
        if (helpers.isParameterDefined('bindEvents', this)) {
            this.bindEvents();
        }
    }
    /**
     * Load config for application
     */
    loadConfig() {
        this.buttons = config.buttons || {};
        this.bot_name = config.bot_name || '';
        this.available_commands = config.available_commands || [];
        this.reply_messages = helpers.processMessages(this, config.reply_messages) || {};
    }
}

/**
 * @export
 */
module.exports = Bot;
